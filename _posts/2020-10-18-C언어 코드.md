#define ARRAYSIZE 100

typedef struct CPB {
	char name[ARRAYSIZE];
	char phoneNumber1[ARRAYSIZE];
	char phoneNumber2[ARRAYSIZE];
	char email[ARRAYSIZE];
} PBOOK;


void insert(PBOOK *p, int n)
{
	scanf("%s", p[n].name);
	scanf("%s", p[n].phoneNumber1);
	scanf("%s", p[n].phoneNumber2);
	scanf("%s", p[n].email);
	return;
}
